\newcommand{\symmtable}[4]{
    \centering
    \renewcommand\tabcolsep{1.5pt}
    %\begin{adjustbox}{max width=\columnwidth}

    \pgfplotstabletypeset[
        col sep=comma,
        font=\scriptsize,
        %%
        every even row/.style={
            before row={\rowcolor[gray]{0.9}}
            },
        every head row/.style={%
            before row={\toprule},
            after row=\midrule
        },
            every last row/.style={
            after row=\bottomrule,
        },      
        %%
        columns/canonical/.style={numeric type, column name={Canonical}, column type={R{#3}}, 
            /pgf/number format/precision=0, /pgf/number format/fixed},
        columns/multilex/.style={numeric type, column name={Allowed}, column type={R{#4}}, 
            /pgf/number format/precision=0, /pgf/number format/fixed},
        columns/symm/.style={numeric type, column name={Sym.\,($\%$)}, column type={R{1.4cm}}, 
            /pgf/number format/precision=2, fixed zerofill},
        columns/eff/.style={numeric type, column name={Eff.\,($\%$)}, column type={R{1.2cm}}, 
            /pgf/number format/precision=2, fixed zerofill},
        %%
        #2
        ]
        {#1}%
    %\end{adjustbox}
}
