\newcommand{\sensorgraph}[1]{
\begin{tikzpicture}[
    sensor/.style={draw,circle,text=black,font=\small,inner sep=0pt,text width=4mm,align=center},
    target/.style={draw,circle,text=black,font=\scriptsize,inner sep=0pt,text width=3mm,align=center,fill=gray!50},
]
    
    \node[sensor] (s1) []	{$s_1$};
    \node[sensor] (s2) [right = 2.7cm of s1.center, anchor=center]	{$s_2$};
    
    \node[sensor] (s3) [below right = 0.9cm and 0.9cm of s1.center, anchor=center]	{$s_4$};
    \node[sensor] (s4) [below right = 0.9cm and 0.9cm of s3.center, anchor=center]	{$s_3$};

    \node[sensor] (s5) [below = 2.7cm of s1.center, anchor=center]	{$s_5$};
    \node[sensor] (s6) [right = 2.7cm of s5.center, anchor=center]	{$s_6$};

    \node[target] (t1) [above right = 0.75cm and 0.6cm of s5.center, anchor=center]	{$t_1$};
    \node[target] (t2) [below left = 0.8cm and 0.9cm of s2.center, anchor=center]	{$t_2$};

    #1
    
\end{tikzpicture}%
}

\newcommand{\sensormatrix}[3]{
\begin{tabular}{ c *{7}{p{0.1cm}} }
      & $s_1$ & $s_2$ & $s_3$ & $s_4$ & $s_5$ & $s_6$ \\
$t_1$ & #1 \\
$t_2$ & #2 \\
\end{tabular}
}
